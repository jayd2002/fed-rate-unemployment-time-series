Creating a Lagged Dataframe

#Creating a Lagged Dataframe. Using 12 lags for econmoic purposes
make_lags <- function(x, K, prefix = "x") {
  out <- map_dfc(0:K, ~dplyr::lag(x, .x))
  names(out) <- paste0(prefix, "_L", 0:K)
  out
}

K <- 12

Xlags <- make_lags(df2$d_fedfunds, K = K, prefix = "dff")
df_m <- bind_cols(df2, Xlags) %>%
  drop_na()

AIC/BIC Test for Lag Selection

#Lag selection using AIC/BIC
fit_by_K <- function(K) {
  Xlags <- make_lags(df2$d_fedfunds, K = K, prefix = "dff")
  dm <- bind_cols(df2, Xlags) %>% drop_na()
  lag_terms <- paste0("dff_L", 0:K, collapse = " + ")
  fml <- as.formula(paste("d_unrate ~", lag_terms, "+ infl"))
  fit <- lm(fml, data = dm)
  tibble(K = K, AIC = AIC(fit), BIC = BIC(fit))
}

ic_tbl <- map_dfr(0:18, fit_by_K)
